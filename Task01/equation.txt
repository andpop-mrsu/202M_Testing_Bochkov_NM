# Чек-лист для тестирования функции решения квадратного уравнения ax^2 + bx + c = 0
# Формат: a, b, c | ожидаемый результат | описание кейса

1) 1, -3, 2 | (1.0, 2.0) | Два различных корня, дискриминант > 0
2) 1, -4, 4 | (2.0, 2.0) | Один корень (два одинаковых), дискриминант = 0
3) 1, 0, -4 | (-2.0, 2.0) | Корни с разными знаками, b = 0
4) 1, 5, 6 | (-3.0, -2.0) | Два отрицательных корня
5) 2, 4, 2 | (-1.0, -1.0) | Один корень с отрицательным значением
6) 1, 0, 0 | (0.0, 0.0) | Единственный корень 0
7) 0, 2, 4 | (-2.0) | Линейное уравнение (a = 0)
8) 0, 0, 5 | None | Нет корней (вырожденное уравнение)
9) 0, 0, 0 | Infinity | Бесконечное множество решений
10) 1, 2, 5 | None | Дискриминант < 0, нет действительных корней
